2 DADA2

This script performs the core bioinformatic processing of 16S rRNA amplicon sequencing data using the DADA2 pipeline (Callahan et al., 2016). The workflow processes raw FASTQ files in parallel chunks for computational efficiency and includes the following steps: (1) primer removal using universal 16S rRNA primers (27F/1492R), (2) quality filtering and length trimming (1000-1600 bp), (3) error rate learning using a pre-computed error model, (4) dereplication of identical sequences, (5) denoising to resolve amplicon sequence variants (ASVs), (6) chimera removal using the consensus method, and (7) filtering to retain only ASVs present in more than one sample. The output is a chimera-free sequence table for each processing chunk that can be merged across all parallel jobs for downstream ecological analysis.

Code 
# Parse command line arguments
args <- commandArgs(trailingOnly=TRUE)
group_index <- as.integer(args[1])      # Group index for parallel processing
fastq_dir   <- args[2]                  # Input directory with S3-mounted FASTQ files
out_dir     <- args[3]                  # Output directory on scratch storage
n_chunks    <- as.integer(args[4])      # Total number of chunks for parallelization
error_file  <- args[5]                  # Pre-computed error model file

# Create output directories for pipeline steps
dir.create(file.path(out_dir, "noprimers"), recursive=TRUE, showWarnings=FALSE)  # Primer-removed files
dir.create(file.path(out_dir, "filtered"), recursive=TRUE, showWarnings=FALSE)   # Filtered sequences
dir.create(file.path(out_dir, "seqtabs"),  recursive=TRUE, showWarnings=FALSE)   # Sequence tables

# List all FASTQ files and split into chunks for parallel processing
fn <- list.files(fastq_dir, pattern="fastq.gz$", full.names=TRUE)
fastq_groups <- split(fn, cut(seq_along(fn), n_chunks, labels=FALSE))
group_files <- fastq_groups[[group_index]]
cat("Group", group_index, "- Number of files:", length(group_files), "\n")

# Remove primers using 16S rRNA universal primers
F27 <- "AGRGTTYGATYMTGGCTCAG"      # Forward primer (27F) for 16S rRNA
R1492 <- "RGYTACCTTGTTACGACTT"     # Reverse primer (1492R) for 16S rRNA
nops <- file.path(out_dir, "noprimers", basename(group_files))  # Output files for primer-removed sequences
mclapply(seq_along(group_files), function(i) {
  removePrimers(group_files[i], nops[i], primer.fwd=F27, primer.rev=dada2:::rc(R1492), orient=TRUE)
}, mc.cores=detectCores())

# Filter and trim sequences by length and quality
filts <- file.path(out_dir, "filtered", basename(group_files))  # Output files for filtered sequences
filterAndTrim(nops, filts, minLen=1000, maxLen=1600, maxN=0, rm.phix=FALSE, multithread=TRUE)

# Load pre-computed error model for DADA2 denoising
err <- readRDS(error_file)

# Dereplicate sequences - collapse identical reads
derep_list <- mclapply(filts, derepFastq, verbose=TRUE, mc.cores=detectCores())
names(derep_list) <- sapply(filts, function(f) tools::file_path_sans_ext(basename(f)))

# Denoise sequences using DADA2 algorithm with error model
dada_list <- dada(derep_list, err=err, BAND_SIZE=32, multithread=TRUE, OMEGA_C=0)
names(dada_list) <- names(derep_list)

# Construct sequence table (ASV table)
seqtab <- makeSequenceTable(dada_list)

# Remove chimeric sequences using consensus method
seqtab_nochim <- removeBimeraDenovo(
  seqtab,
  method = "consensus",
  multithread = TRUE,
  verbose = TRUE
)

# Filter ASVs: keep only those present in more than one sample
keep <- colSums(seqtab_nochim > 0) > 1
seqtab_nochim <- seqtab_nochim[, keep]

# Save chimera-filtered sequence table for this chunk
saveRDS(seqtab_nochim, file=file.path(out_dir, "seqtabs", paste0("seqtab_group", group_index, "_nochim.rds")))

cat("Chunk", group_index, "completed\n")

3 Merge seqtabs

This script combines sequence tables from parallel DADA2 processing chunks and performs taxonomic classification. The workflow: (1) loads all individual sequence tables (seqtab_group*.rds) generated from parallel processing, (2) merges them into a unified ASV table using DADA2’s mergeSequenceTables function, and (3) assigns taxonomy to amplicon sequence variants (ASVs) using a reference database (e.g., SILVA). The output includes both the merged sequence table and corresponding taxonomic classifications, providing the comprehensive ASV table required for downstream ecological and statistical analyses.

Code 
# Parse command line arguments
args <- commandArgs(trailingOnly=TRUE)
seqdir <- args[1]   # Directory containing seqtab_group*.rds files from parallel processing
outfile <- args[2]  # Output file for merged sequence table
taxafile <- args[3] # Reference database for taxonomy (e.g., SILVA fasta file)

# 1. Load all individual sequence tables from parallel chunks
files <- list.files(seqdir, pattern="seqtab_group.*rds$", full.names=TRUE)
cat("Merging", length(files), "sequence tables...\n")
seqtabs <- lapply(files, readRDS)  # Read all chunk files

# 2. Merge sequence tables into a unified ASV table
seqtab_all <- do.call(mergeSequenceTables, seqtabs)  # Combine all chunks
saveRDS(seqtab_all, outfile)  # Save merged sequence table
cat("Merged sequence table saved:", outfile, "\n")

# 3. Taxonomic classification using reference database
cat("Taxonomic assignment with", taxafile, "...\n")
taxa <- assignTaxonomy(seqtab_all, taxafile, multithread=TRUE, tryRC=TRUE)  # Assign taxonomy to ASVs

# Save taxonomic classifications
taxa_out <- sub("\\.rds$", "_taxa.rds", outfile)  # Generate taxonomy output filename
saveRDS(taxa, taxa_out)  # Save taxonomy results
cat("Taxonomy saved:", taxa_out, "\n")

4 Generate count table

sequence variants with their taxonomic classifications. The workflow: (1) constructs a phyloseq object combining ASV abundances and taxonomy, (2) filters out potential contaminants (mitochondrial and chloroplast sequences), (3) assigns unique identifiers to amplicon sequence variants (ASVs), and (4) exports a comprehensive abundance matrix in wide format with samples as columns and ASVs as rows, including full taxonomic lineage. The resulting CSV file provides the clean, formatted count matrix ready for diversity analyses, differential abundance testing, and other ecological investigations.

Code 
# Parse command line arguments
args <- commandArgs(trailingOnly = TRUE)
seqtab_file <- args[1]  # Merged sequence table from previous step (e.g., "output/seqtab_all.rds")
tax_file    <- args[2]  # Taxonomic assignments file (e.g., "output/seqtab_all_taxa.rds")  
out_csv     <- args[3]  # Output file for final abundance matrix (e.g., "output/ps.count_abundance.full.csv")

# Load merged sequence table and taxonomy
seqtab <- readRDS(seqtab_file)  # ASV abundance table
taxa   <- readRDS(tax_file)     # Taxonomic classifications

# Create phyloseq object combining abundances and taxonomy
ps <- phyloseq(
  otu_table(seqtab, taxa_are_rows = FALSE),  # ASV counts (samples × features)
  tax_table(taxa),                           # Taxonomic table
  sample_data(data.frame(SampleID = rownames(seqtab), row.names = rownames(seqtab)))  # Sample metadata
)

# Filter out potential contaminants: chloroplast and mitochondrial sequences
ps_filtered <- subset_taxa(ps, Order != "Chloroplast" & Family != "Mitochondria")

# Add DNA sequences as ASV identifiers and assign unique ASV names
dna <- Biostrings::DNAStringSet(taxa_names(ps_filtered))  # Extract DNA sequences
names(dna) <- taxa_names(ps_filtered)
ps_filtered <- merge_phyloseq(ps_filtered, dna)           # Add sequences to phyloseq object
taxa_names(ps_filtered) <- paste0("ASV", seq(ntaxa(ps_filtered)))  # Rename with ASV1, ASV2, etc.

# Generate final abundance matrix in wide format
psmelt(ps_filtered) %>%                                    # Convert to long format
  arrange(OTU) %>%
  rename(ASV = OTU) %>%                                    # Rename column
  select(ASV, Kingdom, Phylum, Class, Order, Family, Genus, Species, Sample, Abundance) %>%  # Select relevant columns
  pivot_wider(
    names_from = Sample,                                   # Convert to sample columns
    values_from = Abundance,                               # Abundance values
    values_fill = 0                                        # Fill missing with zeros
  ) %>%
  write.csv(out_csv, row.names = FALSE)                    # Export as CSV

cat("Complete abundance matrix saved to:", out_csv, "\n")
